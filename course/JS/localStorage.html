LocalStorage

El LocalStorage tiene la función de almacenar información de manera local indefinidamente hasta que se limpien los datos del navegador.

Inserta información en el LocalStorage
localStorage.setItem("key_name", value)
Extrae información del LocalStorage
let information = localStorage.getItem("key_name")

Ejemplo: Almacenamiento y visualización de datos

    <form>
    <label for="nombre">Nombre</label>
    <input type="text" id="nombre">
    <label for="apellido">Apellido</label>
    <input type="text" id="apellido">
    <label for="fechaNacimiento">fechaNacimiento</label>
    <input type="text" id="fechaNacimiento">
</form>
<ul type="listar-usuarios"></ul>

    let nombre = document.getElementById("nombre")
let apellido = document.getElementById("apellido")
let fechaNacimiento = document.getElementById("fechaNacimiento")
let botonEnviar = document.getElementById("botonEnviar")

class Usuarios{
    constructor(nombre, apellido, fechaNacimiento){
        this.nombre = nombre
        this.apellido = apellido
        this.fechaNacimiento = fechaNacimiento
    }
}

botonEnviar.addEventListener("click", function(){
    let valorNombre = nombre.value
    let valorApellido = apellido.value
    let valorFechaNacimiento = fechaNacimiento.value

    const usuario new Usuarios(valorNombre, valorApellido, valorFechaNacimiento)
    let usuarios = []

    let localUsuarios = localStorage.getItem("usuarios")

    let(localUsuarios){
        usuarios = JSON.parse(localUsuarios)
    }
    usuarios.push(usuario)

    localStorage.setItem("usuarios", JSON.stringify(usuarios))
    alert("Usuario registrado")
})

const mostrarUsuarios = function(){
    let usuarios = []
    let localUsuarios = localStorage.getItem("usuarios")
    if (localUsuarios){
        usuarios = JSON.parse(localUsuarios)
    }
    let listaUsuarios = document.getElementById("listar-usuarios")
    usuarios.forEach(usuario => {
        //createElement crea un elemento HTML desde JavaScript
        let datoUsuario = document.createElement("li")
        datoUsuario.textContent = `Nombre: ${usuario.nombre} - Apellido: ${usuario.apellido} - Fecha nacimiento: ${usuario.fecha}`
        //appendChild permite agregar un elemento HTML desde JavaScript
        listaUsuarios.appendChild(datoUsuario)
    })
}

mostrarUsuarios()
