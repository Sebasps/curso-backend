DOM (Document Object Model)

El DOM permite modificar todos los objetos que tenga en algún documento HTML por medio de los selectores.

Selectores

Selector Descripción Identificación Utilización
Etiqueta Hace referencia a cualquier elemento que posea la misma etiqueta. <tag> document.getElementsByTagName("tag")
Class Las clases se pueden utilizar en multiples elementos. <tag class="class_name"> document.getElementsByClassName("class_name")
ID Un mismo ID solo lo debe tener un elemento por documento. <tag id="id_name"> document.getElementById("id_name")

Ejemplo: calculadora




<form>
    <label>Ingrese el numero 1</label>
    <input type="number" class="numero"><br>
    <label>Ingrese el numero 2</label>
    <input type="number" class="numero"><br>
    <input type="button" id="suma" value="sumar">
    <input type="button" id="resta" value="restar">
    <input type="button" id="multiplicacion" value="multiplicar">
    <input type="button" id="division" value="dividir">
</form>
<p id="resultadoCalculadora"></p>

//Captura de elementos HTML por medio del DOM y su selector
//Document hace referencia al documento HTML
let botonSuma = document.getElementById("suma")
let botonResta = document.getElementById("resta")
let botonMultiplicacion = document.getElementById("multiplicacion")
let botonDivision = document.getElementById("division")
let resultado

function operacion(operador){
    //Value captura los valores de entrada de un formulario por medio del selector
    //Para acceder a un elemento en especifico se necesita especificar la posición
    let numeroUno = document.getElementsByClassName("numero")[0].value
    let numeroDos = document.getElementsByClassName("numero")[1].value
    if (operador=="suma"){
        //Con los valores obtenidos se realiza el evento predefinido
        resultado = parseInt(numeroUno)+parseInt(numeroDos)
    }
    else if (operador=="resta"){
        resultado = parseInt(numeroUno)-parseInt(numeroDos)
    }
    else if (operador=="multiplicacion"){
        resultado = parseInt(numeroUno)*parseInt(numeroDos)
    }
    else if (operador=="division"){
        resultado = parseInt(numeroUno)/parseInt(numeroDos)
    }
    let impresionResultado = document.getElementById("resultadoCalculadora")
    impresionResultado.textContent = `El resultado es: ${resultado}`
}
//AddEventListener crea eventos
botonSuma.addEventListener("click",function(){
    operacion("suma")
})
botonResta.addEventListener("click",function(){
    operacion("resta")
})
botonMultiplicacion.addEventListener("click",function(){
    operacion("multiplicacion")
})
botonDivision.addEventListener("click",function(){
    operacion("division")
})
